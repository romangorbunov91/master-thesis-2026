\section{Алгоритм с изменяемой длительностью измерительного окна, кратной периоду инжектируемого сигнала}\label{sec:fr_optim}

\needspace{1\baselineskip}
Принципиальные особенности рассматриваемого алгоритма:
\begin{itemize}
    \item Длительность ${T_{h}}$ измерительного окна кратна периоду инжектируемого сигнала (${T^{(k)}_{h} = L^{(k)}T^{(k)}_{z}}$, ${T^{(k)}_{h}}$ = var).
    \item Период дискретизации ${T_{s}}$ кратен периоду ${T_{d}}$ выполнения операций в системе (${T^{(k)}_{s} = M^{(k)} T_d}$); коэффициент децимации ${M^{(k)} = var}$, ${M \in \mathbb{N}_{> 0}}$.
    \item Количество точек ${N \in \mathbb{N}_{> 0}}$ дискретной последовательности, по которым вычисляются коэффициенты разложения в выражении \eqref{eqn:DFT}, соответствует каждому отдельному значению ${T^{(k)}_{z}}$ (${N^{(k)} = var}$).
    \begin{equation*}
        N^{(k)} = \cfrac{T^{(k)}_{h}} {T^{(k)}_{s}}.
    \end{equation*}
\end{itemize}

В отличии от алгоритма \ref{sec:Th_const}, длительность измерительного окна варьируется (рисунок~\ref{fig:Th_var_N_var}):
\begin{equation}\label{eqn:windowlength}
    T^{(k)}_{h} = L^{(k)} \mathord{\cdot} T^{(k)}_{z} = L^{(k)} \mathord{\cdot} R^{(k)} \mathord{\cdot} T_{d} = \underbrace{L^{(k)} \mathord{\cdot} Q^{(k)}}_{N^{(k)}} \mathord{\cdot} \underbrace{M^{(k)} \mathord{\cdot} T_d}_{T^{(k)}_{s}},
\end{equation}
где ${R^{(k)} \in \mathbb{R}}$ -- соотношение между периодом инжектируемого сигнала и периодом выполнения операций в системе:
\begin{equation*}
    R^{(k)} = \cfrac{T^{(k)}_z} {T_d} = Q^{(k)} \mathord{\cdot} M^{(k)},
\end{equation*}
${Q^{(k)} \in \mathbb{R}}$.

Аналогично алгоритму \ref{sec:Th_const}, соотношение между периодом инжектируемого сигнала и периодом дискретизации может принимать дробные значения. В совокупности с изменяемой длительностью измерительного окна это обеспечивает на интервале ${f \in \left(0; {f_d / 2} \right)}$ сколь угодно много потенциальных значений частот ${f_z}$ инжектируемого сигнала, что \textbf{является главной отличительной особенностью алгоритма}.

Из равенства \eqref{eqn:windowlength}
\begin{equation}\label{eqn:L_ratio}
    L^{(k)} = \frac{N^{(k)} \mathord{\cdot} M^{(k)}}{R^{(k)}}.
\end{equation}

В связи с тем, что частота инжектируемого сигнала кратна частоте первой гармоники в спектре
\begin{equation*}
    f^{(k)}_{z} = L^{(k)} \mathord{\cdot} f^{(k)}_{h} = \cfrac{L^{(k)}}{T^{(k)}_{h}},
\end{equation*}
множитель ${L^{(k)}}$ ограничен диапазоном
\begin{equation*}
    1 \leq L^{(k)} < \frac{N^{(k)}}{2},
\end{equation*}
что, с учетом \eqref{eqn:L_ratio}, накладывает ограничение
\begin{equation*}
    M^{(k)} < \frac{R^{(k)}}{2}.
\end{equation*}

Таким образом, для каждой $k$-й частоты ${f_{z}}$ задача сводится к поиску наборов ${\mathbf{s}_{(p)} \in \mathbb{N}_{> 0}^3}$ натуральных чисел ${N, M, L}$, при которых обеспечивается частота ${f_{z(p)}}$ с заданной степенью приближения ${\Delta\epsilon^{*}_{f}}$ к ${f_{z}}$
\begin{equation*}
    \left| f_{z(p)} - f_{z} \right| = \Delta\epsilon_{f(p)} < \Delta\epsilon^{*}_{f},
\end{equation*}
при ограничениях
\begin{equation*}
        N_\text{min} \leq N_{(p)} \leq N_\text{max}, \quad M_{(p)} < R/2.
\end{equation*}

Оптимальным считается такой набор ${\mathbf{s}^{*}}$, в котором параметр ${N}$ минимален из всех ${\mathbf{s}_{(p)}}$, отвечающих условию
\begin{equation}\label{eqn:freqtol}
    \Delta\epsilon_{f(p)} < \Delta\epsilon^{*}_{f}.
\end{equation}
В случае, когда ни один набор параметров не удовлетворяет условию \eqref{eqn:freqtol}, оптимальным считается набор ${\mathbf{s}^{*}_{(p)}}$, соответствующий минимальному отклонению ${\Delta\epsilon_{f(p)}}$. Таким образом, минимум количества точек $N$ обеспечивает минимальную продолжительность измерения частотной характеристики.
