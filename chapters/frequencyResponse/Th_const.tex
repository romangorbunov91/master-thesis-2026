\section{Алгоритм с неизменными длительностью измерительного окна и количеством точек}\label{sec:Th_const}

Данный алгоритм характеризуется тем, что в серии из $K$ изменений периода ${T_{z}}$ инжектируемого сигнала (${T^{(k)}_{z}}$, ${k \in \left[1;K \right]}$):
\begin{itemize}
    \item Длительность ${T_{h}}$ измерительного окна неизменна и равна наибольшему периоду инжектируемого сигнала (${T^{(k)}_{h} = T^{max}_{z}}$, ${T^{(k)}_{h} = const}$).
    \item Период дискретизации ${T_{s}}$ одинаков для всех значений ${T_{z}}$ (${T^{(k)}_{s} = const}$) и кратен периоду ${T_{d}}$ выполнения операций в системе (${T^{(k)}_{s} = M \cdot T_d}$); коэффициент децимации ${M = const}$, ${M \in \mathbb{N}}$.
    \item Количество точек ${N \in \mathbb{N}}$ дискретной последовательности, по которым вычисляются коэффициенты разложения в выражении \eqref{eqn:DFT}, неизменно для всех значений ${T^{(k)}_{z}}$ (${N^{(k)} = const}$),
    \begin{equation*}
        N^{(k)} = \cfrac{T^{(k)}_{h}} {T^{(k)}_{s}}.
    \end{equation*}
\end{itemize}

Математическая обработка в соответствии с \eqref{eqn:DFT} накладывает ограничение на период ${T_{z}}$ инжектируемого сигнала -- требуется, чтобы измерительное окно ${T_{h}}$ вмещало целое число периодов ${T_{z}}$ (рисунок~\ref{fig:Th_const}), т.е. для каждого значения ${T^{(k)}_{z}}$ должно выполняться
\begin{equation}\label{eqn:T_h_const}
    T^{(k)}_{h} = L^{(k)} \cdot T^{(k)}_{z},
\end{equation}
где ${L^{(k)} \in \mathbb{N}}$.

    \centering

    \caption{Период инжектируемого сигнала при ${T^{(k)}_{h} = const}$}
    \label{fig:Th_const}
\end{figure}

В соответствии с \eqref{eqn:T_h_const}, частота ${f^{(k)}_{z}}$ инжектируемого сигнала потенциально принимает значения
\begin{equation*}
    f^{(k)}_{z} = L^{(k)} \cfrac{1}{T_{h}} = L^{(k)} \Delta f_{z},
\end{equation*}
где ${\Delta f_{z}}$ -- шаг приращения частоты инжектируемого сигнала

В связи с ${\Delta f_{z} = const}$ потенциальные значения частот ${f^{(k)}_{z}}$ равномерно распределены на интервале возможных значений, что \textbf{является главной отличительной особенностью алгоритма} в сравнении с алгоритмами \ref{sec:N_var}, \ref{sec:N_const}.

Составляющей с частотой ${f^{(k)}_{z}}$ соответствуют коэффициенты ${X^{(k)}\left[l\right]}$, ${Y^{(k)}\left[l\right]}$ при разных значениях ${L^{(k)}}$, определяемых выражением \eqref{eqn:complexFreqResp}.

Таким образом, изложенному алгоритму измерения частотных характеристик с последовательной сменой частоты ${f^{(k)}_{z}}$ инжектируемого сигнала в серии из $K$ измерений \textbf{характерны следующие особенности}:
\begin{enumerate}
    \item Минимально возможная частота ${f_{z}}$ напрямую определяется максимальным значением измерительного окна ${T_{h}}$, которое ограничено сверху только объемом памяти для хранения массивов-откликов ${x\left[n\right]}$, ${y\left[n\right]}$.
    
    \item Максимально возможная частота ${f_{z}}$ в пределе составляет ${f_{s} / 3}$.
    
    \item Требуемый объем памяти для хранения массивов-откликов ${x^{(k)}\left[n\right]}$, ${y^{(k)}\left[n\right]}$ одинаков для всех частот ${f^{(k)}_{z}}$ инжектируемого сигнала.

    \item Период $T_{z}$ инжектируемого сигнала потенциально принимает значения от ${T^{min}_{z} = 3 T_{s}}$ до ${T^{max}_{z} = T_{h}}$, но при условии ${T_{h} / T_{s} \in \mathbb{N}}$.
    
    \item Шаг приращения частоты ${f_{z}}$ инжектируемого сигнала (разрешающая способность): ${\Delta f_{z} = 1 / T_{h}}$.
\end{enumerate}
