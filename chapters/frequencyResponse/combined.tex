\section{Алгоритм с оптимальным набором параметров}\label{sec:fr_optim}

Алгоритм позволяет найти такой набор параметров, при котором частотная характеристика измеряется за минимальное время при заданной точности воспроизведения частоты.

Данный алгоритм характеризуется тем, что в серии из $K$ изменений периода ${T_{z}}$ инжектируемого сигнала ${\left({T^{(k)}_{z}}, k \in \left[1;K \right]\right) }$:
\begin{itemize}
    \item Длительность ${T_{h}}$ измерительного окна кратна периоду инжектируемого сигнала (${T^{(k)}_{h} = L^{(k)}T^{(k)}_{z}}$, ${T^{(k)}_{h}}$ = var).
    \item Период дискретизации ${T_{s}}$ кратен периоду ${T_{d}}$ выполнения операций в системе (${T^{(k)}_{s} = M^{(k)} T_d}$); коэффициент децимации ${M^{(k)} = var}$, ${M \in \mathbb{N}_{> 0}}$.
    \item Количество точек ${N \in \mathbb{N}_{> 0}}$ дискретной последовательности, по которым вычисляются коэффициенты разложения в выражении \eqref{eqn:DFT}, соответствует каждому отдельному значению ${T^{(k)}_{z}}$ (${N^{(k)} = var}$).
    \begin{equation*}
        N^{(k)} = \cfrac{T^{(k)}_{h}} {T^{(k)}_{s}}.
    \end{equation*}
\end{itemize}

Длительность измерительного окна
\begin{equation}\label{eqn:windowlength}
    T^{(k)}_{h} = L^{(k)} \cdot T^{(k)}_{z} = L^{(k)} \cdot R^{(k)} \cdot T_{d} = \underbrace{L^{(k)} \cdot Q^{(k)}}_{N^{(k)}} \cdot \underbrace{M^{(k)} \cdot T_d}_{T^{(k)}_{s}},
\end{equation}
где ${R^{(k)} \in \mathbb{R}}$ -- соотношение между периодом инжектируемого сигнала и периодом выполнения операций в системе
\begin{equation*}
    R^{(k)} = \cfrac{T^{(k)}_{z}} {T_{d}} = Q^{(k)} \cdot M^{(k)},
\end{equation*}
${Q^{(k)} \in \mathbb{R}}$.

\textbf{Главной отличительной особенностью алгоритма} является то, что соотношение между периодом инжектируемого сигнала и периодом дискретизации может принимать дробные значения. Это означает, что на интервале ${f \in \left(0; {f_{d} / 2} \right)}$ потенциальные значения частот ${f_{z}}$ инжектируемого сигнала принимают сколь угодно много значений, ограниченное исключительно объемом памяти для хранения массивов-откликов ${x\left[n\right]}$, ${y\left[n\right]}$.

Из равенства \eqref{eqn:windowlength}
\begin{equation}\label{eqn:L_ratio}
    L^{(k)} = \frac{N^{(k)} \cdot M^{(k)}}{R^{(k)}}.
\end{equation}

В связи с тем, что частота инжектируемого сигнала
\begin{equation*}
    f^{(k)}_{z} = L^{(k)} \cdot f^{(k)}_{h} = \cfrac{L^{(k)}}{T^{(k)}_{h}}
\end{equation*}
кратна частоте первой гармоники в спектре, множитель ${L^{(k)}}$ ограничен диапазоном
\begin{equation*}
    1 \leq L^{(k)} < \frac{N^{(k)}}{2},
\end{equation*}
что, с учетом \eqref{eqn:L_ratio}, накладывает ограничение
\begin{equation*}
    M^{(k)} < \frac{R^{(k)}}{2}.
\end{equation*}

Таким образом, для каждой $k$-й частоты ${f_{z}}$ задача сводится к поиску наборов ${\mathbf{s}_{(p)} \in \mathbb{N}_{> 0}^3}$ целых чисел $N, M, L$, при которых обеспечивается частота ${f^{*}_{z}}$ с заданной степенью приближения ${\epsilon_{f}}$ к ${f_{z}}$:
\begin{equation}\label{eqn:freqtol}
    \left| f^{*}_{z} - f_{z} \right| = \epsilon_{f} < \epsilon^{*}_{f},
\end{equation}
при ограничениях
\begin{equation*}
    \begin{aligned}
        N_\text{min} \leq &N^{(k)} \leq N_\text{max}, \\
        M^{(k)} & < \frac{R^{(k)}}{2}.
    \end{aligned}
\end{equation*}

Набор ${\mathbf{s}_{(p)}}$ считается оптимальным ${\mathbf{s}^{*}_{(p)}}$ при условии
\begin{equation*}
    N^{*(k)} = \min \left\{ N^{(k)} \;\middle|\; N^{(k)} \in \left[ N_\text{min}; N_\text{max} \right] \;\middle|\; \epsilon^{(k)}_{f} < \epsilon^{*}_{f} \right\}.
\end{equation*}

В случае, когда ни один набор параметров не удовлетворяет условию \eqref{eqn:freqtol}, оптимальным считается набор ${\mathbf{s}^{*}_{(p)}}$ при условии
\begin{equation*}
    N^{*(k)}: \min \left\{ \epsilon^{(k)}_{f} \right\}.
\end{equation*}

Последовательность нахождения оптимального набора параметров:
\begin{enumerate}
    \item Вычисляется коэффициент
    \item Перебираются значения $N$ из диапазона ${N_\text{min}}$, ${N_\text{max}}$.
\end{enumerate}
