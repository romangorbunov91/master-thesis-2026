\chapter{Разработка алгоритмов идентификации параметров макромоделей}
Данный раздел относится к задаче автоматизированного контроля. Стоит задача разработать универсальную архитектуру, способную идентифицировать отклонения по частотным характеристикам. Частотные характеристики измеряются, либо подгружаются извне.

\section{Архитектура модели}

В основу модели положен автоэнкодер.
На вход модели подается 1D-массив размерностью $[B, C_{in}, Lenght]$. На выходе формируется бинарная маска, состоящая из $C_{out}$-каналов, каждый из которых соответствует одному из объектов к идентификации.

\section*{Notation}

Input tensor: $\mathbf{X} \in \R^{B \times C_{in} \times L}$ (batch $\times$ channels $\times$ sequence length)
Convolution: $\Conv$, element-wise operations: $\oplus$ (add), $\odot$ (multiply)
Activations: $\GELU(\cdot)$, $\ReLU(\cdot)$, $\sigmoid(\cdot) = (1+e^{-x})^{-1}$
Normalization: $\GN(\cdot)$ denotes GroupNorm with 1 group (LayerNorm over spatial dimension)
Pooling/Upsampling: $\Pool(\cdot)$ max-pooling (kernel=2, stride=2); $\Upsample(\cdot)$ linear interpolation

\section{Residual Convolutional Block}
\label{sec:resconv}

For input $\mathbf{x} \in \R^{B \times C_{in} \times L}$, output $\mathbf{y} \in \R^{B \times C_{out} \times L}$:
\begin{comment}

\begin{align}
    \mathbf{z}_1 &= \GELU\!\Big(\GN\big(\mathbf{W}^{(1)}_1 \Conv \mathbf{x}\big)\Big) \label{eq:conv1} \\
    \mathbf{z}_2 &= \Dropout\!\Big(\GELU\!\Big(\GN\big(\mathbf{W}^{(1)}_2 \Conv \mathbf{z}_1\big)\Big)\Big) \label{eq:conv2} \\
    \mathbf{s} &= 
    \begin{cases}
        \mathbf{W}_{skip} \Conv \mathbf{x}, & C_{in} \neq C_{out} \\
        \mathbf{x}, & C_{in} = C_{out}
    \end{cases} \label{eq:skip} \\
    \mathbf{y} &= \GELU\!\Big(\mathbf{z}_2 \oplus \mathbf{s}\Big) \label{eq:resout}
\end{align}

where $\mathbf{W}^{(1)}_1, \mathbf{W}^{(1)}_2 \in \R^{C_{out} \times C_{in} \times 3}$ (kernel size 3, padding 1, bias=False).

\section{Attention Gate (Attention U-Net Style)}
\label{sec:attn gate}

Given gating signal $\mathbf{g} \in \R^{B \times F_g \times L_g}$ and skip feature $\mathbf{x} \in \R^{B \times F_l \times L_x}$:


\begin{align}
    \mathbf{x}' &= \Interp(\mathbf{x}; \text{size}=L_g) \quad \text{if } L_x \neq L_g \label{eq:interp} \\
    \mathbf{h}_g &= \GN\big(\mathbf{W}_g \Conv \mathbf{g}\big), \quad \mathbf{W}_g \in \R^{F_{int} \times F_g \times 1} \label{eq:wg} \\
    \mathbf{h}_x &= \GN\big(\mathbf{W}_x \Conv \mathbf{x}'\big), \quad \mathbf{W}_x \in \R^{F_{int} \times F_l \times 1} \label{eq:wx} \\
    \boldsymbol{\psi} &= \sigmoid\!\Big(\mathbf{W}_{\psi} \Conv \ReLU\big(\mathbf{h}_g \oplus \mathbf{h}_x\big)\Big), \quad \mathbf{W}_{\psi} \in \R^{1 \times F_{int} \times 1} \label{eq:psi} \\
    \mathbf{y} &= \mathbf{x}' \odot \boldsymbol{\psi} \in \R^{B \times F_l \times L_g} \label{eq:attnout}
\end{align}

The attention map $\boldsymbol{\psi} \in [0,1]^{B \times 1 \times L_g}$ provides spatial reweighting of skip connections.

\section{Transformer Bottleneck}
\label{sec:transformer}

For input $\mathbf{x} \in \R^{B \times C \times L}$:



    \tilde{\mathbf{x}} &= \GN(\mathbf{x}) \label{eq:norm1} \\
    \mathbf{X}_t &= \tilde{\mathbf{x}}^\top \in \R^{B \times L \times C} \label{eq:transpose} \\
    \mathbf{A} &= \MHSA(\mathbf{X}_t, \mathbf{X}_t, \mathbf{X}_t) \label{eq:mhsa} \\
    \mathbf{x}' &= \mathbf{x} \oplus \mathbf{A}^\top \label{eq:resattn} \\
    \tilde{\mathbf{x}}' &= \GN(\mathbf{x}') \label{eq:norm2} \\
    \mathbf{M} &= \MLP\big((\tilde{\mathbf{x}}')^\top\big)^\top, \quad 
    \MLP(\mathbf{z}) = \mathbf{W}_2 \cdot \Dropout\!\Big(\GELU\big(\mathbf{W}_1 \mathbf{z}\big)\Big) \label{eq:mlp} \\
    \mathbf{y} &= \mathbf{x}' \oplus \mathbf{M} \label{eq:transformerout}


    \noindent where Multi-Head Self-Attention with $h$ heads:
    \begin{equation}
        \MHSA(\mathbf{Q},\mathbf{K},\mathbf{V}) = \Concat_{i=1}^h \Big[ \Softmax\!\Big(\frac{\mathbf{Q}_i\mathbf{K}_i^\top}{\sqrt{d_k}}\Big) \mathbf{V}_i \Big] \mathbf{W}^O
    \end{equation}
    with $\mathbf{Q}_i = \mathbf{X}_t \mathbf{W}^Q_i$, $\mathbf{K}_i = \mathbf{X}_t \mathbf{W}^K_i$, $\mathbf{V}_i = \mathbf{X}_t \mathbf{W}^V_i$, and $d_k = C/h$.

    \section{Upsampling Block}
    \label{sec:upsample}

    \begin{equation}
        \mathbf{y} = \GELU\!\Big(\GN\big(\mathbf{W}_{up} \Conv \Upsample(\mathbf{x})\big)\Big), \quad \mathbf{W}_{up} \in \R^{C_{out} \times C_{in} \times 3}
    \end{equation}

    \section{Full Architecture}
    \label{sec:full}

    Let feature dimensions be $\mathbf{f} = [f_0, f_1, f_2, f_3, f_4]$.

    \subsection*{Encoder}
    \begin{align}
        \mathbf{e}_1 &= \text{ResConvBlock}_{C_{in} \to f_0}(\mathbf{X}) \\
        \mathbf{e}_2 &= \text{ResConvBlock}_{f_0 \to f_1}\big(\Pool(\mathbf{e}_1)\big) \\
        \mathbf{e}_3 &= \text{ResConvBlock}_{f_1 \to f_2}\big(\Pool(\mathbf{e}_2)\big) \\
        \mathbf{e}_4 &= \text{ResConvBlock}_{f_2 \to f_3}\big(\Pool(\mathbf{e}_3)\big)
    \end{align}

    \subsection*{Bottleneck}
    \begin{align}
        \mathbf{b}_0 &= \GELU\!\Big(\GN\big(\mathbf{W}_{proj} \Conv \Pool(\mathbf{e}_4)\big)\Big), \quad \mathbf{W}_{proj} \in \R^{f_4 \times f_3 \times 1} \\
        \mathbf{b} &= \text{TransformerBottleneck}_{f_4}(\mathbf{b}_0)
    \end{align}

    \subsection*{Decoder (with Attention-Gated Skips)}
    For $i \in \{4,3,2,1\}$ descending, with $\mathbf{d}_5 \triangleq \mathbf{b}$:
    \begin{align}
        \mathbf{g}_i &= \text{UpSample}_{f_i \to f_{i-1}}(\mathbf{d}_{i+1}) \\
        \hat{\mathbf{e}}_i &= \text{AttentionGate}_{F_g=f_{i-1},\, F_l=f_{i-1},\, F_{int}=f_{i-1}/2}(\mathbf{g}_i, \mathbf{e}_i) \\
        \mathbf{d}_i &= \text{ResConvBlock}_{2f_{i-1} \to f_{i-1}}\Big(\Concat[\mathbf{g}_i, \hat{\mathbf{e}}_i]\Big)
    \end{align}

    \subsection*{Output Heads}
    \begin{align}
        \mathbf{y}_{\text{main}} &= \mathbf{W}_{final} \Conv \mathbf{d}_1, \quad \mathbf{W}_{final} \in \R^{C_{out} \times f_0 \times 1} \\
        \mathbf{y}^{(k)}_{\text{ds}} &= \mathbf{W}^{(k)}_{ds} \Conv \mathbf{d}_{k+1}, \quad k \in \{0,1,2,3\} \quad \text{(deep supervision)}
    \end{align}

    \noindent \textbf{Forward output:}
    \begin{equation}
        \text{Output} = 
        \begin{cases}
            \big(\mathbf{y}_{\text{main}},\; [\mathbf{y}^{(0)}_{\text{ds}}, \mathbf{y}^{(1)}_{\text{ds}}, \mathbf{y}^{(2)}_{\text{ds}}, \mathbf{y}^{(3)}_{\text{ds}}]\big), & \text{if training \& deep\_supervision} \\
            \mathbf{y}_{\text{main}}, & \text{otherwise}
        \end{cases}
    \end{equation}

    \section{Loss Function (with Deep Supervision)}
    \label{sec:loss}

    \begin{equation}
        \mathcal{L} = \lambda_0 \cdot \ell\big(\mathbf{y}_{\text{main}}, \mathbf{Y}\big) + \sum_{k=0}^{3} \lambda_{k+1} \cdot \ell\big(\mathbf{y}^{(k)}_{\text{ds}}, \mathbf{Y}\big)
    \end{equation}
    where $\ell(\cdot)$ is a task-specific loss (e.g., cross-entropy, Dice), and $\lambda_k$ are weighting coefficients.

    \section*{Summary Diagram (Mathematical Flow)}

    \begin{equation*}
    \boxed{
    \begin{aligned}
    &\text{\textbf{Encode:}} && \{\mathbf{e}_i\}_{i=1}^4 \gets \text{Encoder}(\mathbf{X}) \\
    &\text{\textbf{Bottleneck:}} && \mathbf{b} \gets \text{Transformer}\big(\text{Proj}(\Pool(\mathbf{e}_4))\big) \\
    &\text{\textbf{Decode:}} && \mathbf{d}_1 \gets \text{Decoder}\big(\mathbf{b}, \{\mathbf{e}_i\}; \text{AttentionGates}\big) \\
    &\text{\textbf{Output:}} && \mathbf{y} = \mathbf{W}_{final} \Conv \mathbf{d}_1 \quad (+\; \text{deep supervision if training})
    \end{aligned}
    }
    \end{equation*}
