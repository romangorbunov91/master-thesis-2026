\newcommand{\setupappendix}{%
  \setcounter{chapter}{0}%
  \renewcommand{\chaptername}{Приложение}%
  \renewcommand{\thechapter}{\cyralph{\value{chapter}}}%
  \renewcommand{\theHchapter}{app\thechapter}%
  
  % Safely override chapter TOC format for appendices
  \addtocontents{toc}{%
    \protect\titlecontents{chapter}[0pt]{\vspace{0pt}\normalfont}%
      {\protect\chaptername\ \thecontentslabel.\quad}{}%
      {\protect\titlerule*[1pc]{.}\protect\contentspage}[]%
  }%
  
  % Reset counters per appendix
  \counterwithin{section}{chapter}%
  \counterwithin{figure}{chapter}%
  \counterwithin{table}{chapter}%
  
  % Chapter heading format for appendices
  \titleformat{\chapter}[display]
    {\normalfont\centering\bfseries\fontsize{14pt}{17pt}\selectfont}
    {}
    {0pt}
    {\chaptername~\thechapter.\hspace{0.25em}}%
}
